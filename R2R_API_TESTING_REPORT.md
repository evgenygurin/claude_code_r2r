```python
import time

def retry_request(func, max_retries=3, backoff=2):
    """Retry logic with exponential backoff"""
    for attempt in range(max_retries):
        try:
            result = func()
            if result.get('status_code') in [200, 201]:
                return result
            elif result.get('status_code') == 429:  # Rate limit
                time.sleep(backoff ** attempt)
                continue
            elif result.get('status_code') >= 500:  # Server error
                time.sleep(backoff ** attempt)
                continue
            else:
                return result  # Client error, don't retry
        except Exception as e:
            if attempt == max_retries - 1:
                raise
            time.sleep(backoff ** attempt)
    return None
```

```python
import time
from collections import deque

class RateLimiter:
    def __init__(self, max_requests, time_window):
        """
        max_requests: –º–∞–∫—Å–∏–º—É–º –∑–∞–ø—Ä–æ—Å–æ–≤
        time_window: –≤—Ä–µ–º–µ–Ω–Ω–æ–µ –æ–∫–Ω–æ –≤ —Å–µ–∫—É–Ω–¥–∞—Ö
        """
        self.max_requests = max_requests
        self.time_window = time_window
        self.requests = deque()

    def allow_request(self):
        now = time.time()

        # –£–¥–∞–ª–∏—Ç—å —Å—Ç–∞—Ä—ã–µ –∑–∞–ø—Ä–æ—Å—ã
        while self.requests and self.requests[0] < now - self.time_window:
            self.requests.popleft()

        if len(self.requests) < self.max_requests:
            self.requests.append(now)
            return True
        return False

    def wait_if_needed(self):
        while not self.allow_request():
            time.sleep(0.1)

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ: max 100 requests per minute
```

```python
from typing import List, Dict

class DocumentPipeline:
    def __init__(self, client):
        self.client = client

    def ingest_batch(self, files: List[str], batch_size: int = 10):
        """–ü–∞–∫–µ—Ç–Ω–∞—è –∑–∞–≥—Ä—É–∑–∫–∞ –¥–æ–∫—É–º–µ–Ω—Ç–æ–≤"""

        for i in range(0, len(files), batch_size):
            batch = files[i:i+batch_size]
            batch_results = []

            for file_path in batch:
                result = self.client.create_document(
                    file_path=file_path,
                    metadata=self._extract_metadata(file_path),
                    ingestion_mode="hi-res"
                )
                batch_results.append(result)

            results.extend(batch_results)
            print(f"Processed {len(results)}/{len(files)} documents")

        return results

    def _extract_metadata(self, file_path: str) -> Dict:
        """–ò–∑–≤–ª–µ—á—å –º–µ—Ç–∞–¥–∞–Ω–Ω—ã–µ –∏–∑ –ø—É—Ç–∏ —Ñ–∞–π–ª–∞"""
        import os
        from datetime import datetime

        filename = os.path.basename(file_path)
        ext = os.path.splitext(filename)[1]
        size = os.path.getsize(file_path)

        return {
            "filename": filename,
            "extension": ext,
            "size_bytes": size,
            "upload_date": datetime.now().isoformat(),
            "path": file_path
        }

    def analyze_document(self, document_id: str):
        """–ê–Ω–∞–ª–∏–∑ –¥–æ–∫—É–º–µ–Ω—Ç–∞"""
        # –ü–æ–ª—É—á–∏—Ç—å —á–∞–Ω–∫–∏ –¥–æ–∫—É–º–µ–Ω—Ç–∞

        # –ò–∑–≤–ª–µ—á—å –æ—Å–Ω–æ–≤–Ω—ã–µ —Ç–µ–º—ã
        themes_query = "What are the main themes and topics in this document?"
        themes = self.client.rag_query(
            query=themes_query,
            search_settings={
                "filters": {"document_id": document_id},
                "limit": 10
            }
        )

        # –°–æ–∑–¥–∞—Ç—å summary
        summary_query = "Provide a concise summary of this document."
        summary = self.client.rag_query(
            query=summary_query,
            search_settings={
                "filters": {"document_id": document_id},
                "limit": 5
            },
            rag_generation_config={
                "temperature": 0.3
            }
        )

        return {
            "document_id": document_id,
            "num_chunks": len(chunks.get('chunks', [])),
            "themes": themes['completion'],
            "summary": summary['completion']
        }
```

```python
import time

def exponential_backoff_retry(func, max_retries=5):
    """Retry with exponential backoff"""
    for attempt in range(max_retries):
        result = func()

        if result.get('status_code') == 429:
            wait_time = 2 ** attempt  # 1, 2, 4, 8, 16 seconds
            print(f"Rate limited. Waiting {wait_time}s...")
            time.sleep(wait_time)
            continue

        return result

    raise Exception("Max retries exceeded")
```
